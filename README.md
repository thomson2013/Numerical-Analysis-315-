     function W = wilkin(n)
     %Computes a nxn Wilkinson matrix
     A = zeros(n,n);
     for i = 1:n
    		A(i,n) = 1;
    		A(i,i) = 1;
    		for j = 1:n
        		if i > j
            		A(i,j) = -1;
        		end
    		end
     end
